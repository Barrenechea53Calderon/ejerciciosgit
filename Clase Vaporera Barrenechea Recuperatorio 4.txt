class Vaporera:
    def __init__(self, marca, capacidad):
        self.marca = marca
        self.capacidad = capacidad
        self.encendida = False

    def encender(self):
        if not self.encendida:
            self.encendida = True
            print("La vaporera est치 encendida.")
        else:
            print("La vaporera ya est치 encendida.")

    def apagar(self):
        if self.encendida:
            self.encendida = False
            print("La vaporera est치 apagada.")
        else:
            print("La vaporera ya est치 apagada.")

    def __str__(self):
        estado = "encendida" if self.encendida else "apagada"
        return f"Vaporera(marca={self.marca}, capacidad={self.capacidad}L, estado={estado})"
